Entity Hooks

OnAirdrop

void OnAirdrop(Vector3 location)
{
    Puts("OnAirdrop works!");
}

function PLUGIN:OnAirdrop(location)
    print("OnAirdrop works!")
end

OnAirdrop: (location) =>
    print "OnAirdrop works!"

OnAirdrop: function(location) {
    print("OnAirdrop works!");
}

def OnAirdrop(self, location):
    print "OnAirdrop works!"
  • Called when an airdrop has been called
  • No return behavior
